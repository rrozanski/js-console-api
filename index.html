console.time / console.timeEnd 

            function fib(num) {
                if (num <= 1) {
                    return 1
                }

                return fib(num - 1) + fib(num - 2);
            }

            console.time('firstResult');
            console.log(fib(5));
            console.timeEnd('firstResult');

            console.log('\n----------\n\n');

            console.time('secondResult');
            console.log(fib(40));
            console.timeEnd('secondResult');

            console.log('\n----------\n\n');
